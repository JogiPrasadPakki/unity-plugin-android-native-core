Examples

Alet Dialog


using AndroidNativeCore;
using UnityEngine
class example:MonoBehaviour{

    void Start(){

    }
    public void ShowAlert()
    {
        AlertDialog alertDialog = new AlertDialog();
        
        alertDialog.build(AlertDialog.THEME_HOLO_DARK)
        .setTitle("Hi")
        .setMessage("This holo dark alertDialog")
        .setNegitiveButtion("Cansel",() => { Toast.show("NegitiveButtion clicked", Toast.LENGTH_SHORT); })
        .setPositiveButtion("Ok", () => { Toast.show("PositiveButtion clicked", Toast.LENGTH_SHORT); })
        .show();
    }
}
